    Result txIsFat32(bool *mode) {
        Result rc = serviceDispatch(&g_tx, 137);
        if (rc == 0xa08) {
            *mode = false;
            return 0;
        } else if (rc == 0) {
            *mode = true;
        }
        return rc;
    }
